## ✅ 问题根源

1. **CSS 样式问题**: 按钮的 `pointer-events` 可能被其他元素遮挡
2. **z-index 层级问题**: 按钮可能被表格或其他元素覆盖

---

### 1. 增强按钮 CSS 样式

```css
.btn { 
  pointer-events: auto;
  position: relative;
  z-index: 100;
  display: inline-block;
}

.action-buttons button {
  pointer-events: auto !important;
  z-index: 100 !important;
}

td button {
  pointer-events: auto !important; 
  z-index: 100 !important;
}
```

### 2. 确保按钮事件绑定正确

```javascript
// 按钮 HTML 生成
<button class="btn btn-warning btn-sm" onclick="toggleProductStatus(${p.id}, 0)">下架</button>
<button class="btn btn-success btn-sm" onclick="toggleProductStatus(${p.id}, 1)">上架</button>
```

### 3. JavaScript 函数实现

```javascript
// 上架/下架功能
async function toggleProductStatus(id, newStatus) {
  const product = products.find(p => p.id === id);
  const action = newStatus === 1 ? '上架' : '下架';
  
  vant.Dialog({
    title: '确认操作',
    message: `确定要${action}商品"${product.name}"吗？`,
    showCancelButton: true,
    confirmButtonText: '确定',
    cancelButtonText: '取消'
  }).then(async () => {
    vant.Toast.loading({ message: `${action}中...`, forbidClick: true });
    
    const res = await fetchWithAuth(`${API_BASE}/admin/products/${id}/status`, {
      method: 'PUT',
      body: JSON.stringify({ status: newStatus })
    });
    
    vant.Toast.clear();
    
    if (res.code === 200) {
      vant.Toast.success(`${action}成功`);
      await loadProducts();
    } else {
      vant.Toast.fail(`${action}失败：` + res.message);
    }
  });
}

// 删除功能
async function deleteProduct(id) {
  const product = products.find(p => p.id === id);
  
  vant.Dialog({
    title: '确认删除',
    message: `⚠️ 确定要删除商品"${product.name}"吗？`,
    showCancelButton: true,
    confirmButtonText: '删除',
    cancelButtonText: '取消',
    confirmButtonColor: '#ee0a24'
  }).then(async () => {
    vant.Toast.loading({ message: '删除中...', forbidClick: true });
    
    const res = await fetchWithAuth(`${API_BASE}/admin/products/${id}`, { 
      method: 'DELETE' 
    });
    
    vant.Toast.clear();
    
    if (res.code === 200) {
      vant.Toast.success('商品已移入回收站');
      await loadProducts();
    } else {
      vant.Toast.fail('删除失败：' + res.message);
    }
  });
}
```

## 📝 使用说明

### 上架/下架商品
1. 点击商品的"上架"或"下架"按钮
2. 弹出确认对话框
3. 点击"确定"执行操作
4. 点击"取消"取消操作
5. 操作成功后自动刷新列表

### 删除商品
1. 点击商品的"删除"按钮
2. 弹出详细确认对话框
3. 点击"删除"移入回收站
4. 点击"取消"取消操作
5. 操作成功后自动刷新列表
